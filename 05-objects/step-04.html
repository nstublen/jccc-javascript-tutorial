Pass by Value

Primitive data types (i.e. Number, String, Boolean) are passed to functions by value. 

        function uppercase(text) {
            // Change the value of "text"
            text = text.toUpperCase();
            return text;
        }

        var test = "hello";
        var upper = uppercase(test);

Pass by Reference

Objects and arrays are passed by reference. 

            function uppercase(text) {
                // Change the value of "text"
                for (var index = 0; index < text.length; ++index) {
                    text[index] = text[index].toUpperCase();
                }
                return text;
            }

            var test = ["hello", "hi", "greetings"];
            var upper = uppercase(test);
