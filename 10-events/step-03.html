Mouse Events

        onclick         // Click an element
        oncontextmenu   // Right-click an element
        ondblclick      // Double-click an element
        onmousedown     // Press a mouse button
        onmouseenter    // Pointer moves onto element
        onmouseleave    // Pointer moves out of element
        onmousemove     // Pointer moves while over an element
        onmouseover     // Pointer moves onto element or its children
        onmouseout      // Pointer moves out of element or its children
        onmouseup       // Release a mouse button
